//! [if_else]
---- simd
<- pf =          (1, 2, -1, 0.5) + i * (0, 1, -1, 3)
<- qf =          (-1, -2, -1, -0) + i * (-0, -1, 1, -2)
-> bs::if_else(b, pf, qf)    = (1, -2, -1, -0) + i * (0, -1, -1, -2)
---- scalar
<- xf =          1 + i * 1
<- yf =          1 + i * 1
-> bs::if_else(true, xf, yf)    = 1 + i * 1
//! [if_else]
